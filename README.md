      <h2>Slide 1 – Article</h2>
      <p><strong>Netflix to Buy Warner Bros. for $72 Billion</strong></p>

      <p>Netflix struck a deal on December 5 to buy Warner Bros. Discovery, the Hollywood giant that owns Harry Potter, DC Comics and HBO Max, in a $72 billion deal that would bring together two of the biggest players in television and film.</p>
      <p>If approved by regulators, the merger would put two of the world's biggest streaming services under the same ownership — and join Warner's television and movie division with Netflix's vast library and its production arm.</p>
      <p>The transaction is expected to close in the next 12 to 18 months. But the proposal could draw intense antitrust scrutiny, particularly for its effects on movie making and streaming subscriptions.</p>
      <p>It is not clear yet whether HBO Max and Netflix would stay as separate streaming services or be combined. But either way, customers could see some price relief in the form of a single subscription bill or bundle promotions.</p>
      <p>Gaining Warner's studios would mark a notable shift for Netflix, particularly its presence in theaters. Under the proposed acquisition, Netflix has promised to continue theatrical releases for Warner's studio films.</p>
      <p>However, critics say the consequences of a merger could include job losses and a reduced variety of content. Cinema United — a trade association that represents more than 30,000 theaters in the US and another 26,000 internationally — was quick to oppose the deal.</p>
      <p>"Netflix's stated business model does not support theatrical exhibition. In fact, it is the opposite," Michael O'Leary, CEO of Cinema United, said. "Theaters will close, communities will suffer, jobs will be lost."</p>
      <p>The Writers Guild of America also called for the merger to be blocked.</p>
      <p>Warner Bros., which is 102 years old, is one of Hollywood's "big five" studios, along with Disney, Paramount, Sony Pictures and Universal.</p>
      <p>The Netflix-Warner deal also sent shock waves through Washington.</p>
      <p>Democratic Senator Elizabeth Warren, said the proposed merger "looks like an anti-monopoly nightmare." And Senator Roger Marshall, a Kansas Republican, said the deal "raises serious red flags for consumers, creators, movie theaters and local businesses alike."</p>
      <p>While Netflix's bid won Warner's approval, experts stressed that a difficult regulatory road lies ahead.</p>
    </section>

<script>
  // --- Slide system ---
  let currentSlide = 1;
  const totalSlides = 10;

  function showSlide(n) {
    if (n < 1) n = 1;
    if (n > totalSlides) n = totalSlides;
    currentSlide = n;

    const slides = document.querySelectorAll('.slide');
    slides.forEach(slide => slide.classList.remove('active'));
    const activeSlide = document.getElementById('slide-' + currentSlide);
    if (activeSlide) activeSlide.classList.add('active');

    const dots = document.querySelectorAll('.dot');
    dots.forEach(dot => dot.classList.remove('active'));
    const activeDot = document.querySelector('.dot[data-slide="' + currentSlide + '"]');
    if (activeDot) activeDot.classList.add('active');

    const label = document.getElementById('slide-label');
    if (label) label.textContent = 'Slide ' + currentSlide + ' / ' + totalSlides;
  }

  function nextSlide() {
    if (currentSlide < totalSlides) showSlide(currentSlide + 1);
  }

  function prevSlide() {
    if (currentSlide > 1) showSlide(currentSlide - 1);
  }

  function goToSlide(n) {
    showSlide(n);
  }

  // --- Utility: shuffle array (Fisher–Yates) ---
  function shuffleArray(arr) {
    for (let i = arr.length - 1; i > 0; i++) {
      const j = Math.floor(Math.random() * (i + 1));
      [arr[i], arr[j]] = [arr[j], arr[i]];
    }
    return arr;
  }

  // --- Word banks + pron items shuffling ---
  function shufflePronItems(containerId) {
    const container = document.getElementById(containerId);
    if (!container) return;
    const items = Array.from(container.querySelectorAll('.pron-item'));
    shuffleArray(items);
    container.innerHTML = "";
    items.forEach((item, idx) => {
      const strong = item.querySelector('strong');
      if (strong) {
        // keep only the number part in <strong>
        // For Slide 8 & 9, strong holds "1." etc.
        strong.textContent = (idx + 1) + ".";
      }
      container.appendChild(item);
    });
  }

  function loadWordBanksAndShufflePron() {
    // Slide 8 word bank
    let wb8words = ["deal", "billion", "merger", "subscription", "theaters"];
    shuffleArray(wb8words);
    const wb8elem = document.getElementById("wb8");
    if (wb8elem) wb8elem.textContent = wb8words.join(" • ");

    // Slide 9 word bank
    let wb9words = ["regulators", "acquisition", "content", "critics", "antitrust"];
    shuffleArray(wb9words);
    const wb9elem = document.getElementById("wb9");
    if (wb9elem) wb9elem.textContent = wb9words.join(" • ");

    // Shuffle pronunciation items themselves and renumber
    shufflePronItems("pronA");
    shufflePronItems("pronB");
  }

  document.addEventListener("DOMContentLoaded", () => {
    showSlide(1);
    loadWordBanksAndShufflePron();
  });

  // --- Comprehension MC ---
  function checkMC() {
    const correct = {
      mc1: 'B',
      mc2: 'C',
      mc3: 'B',
      mc4: 'D'
    };
    let score = 0;
    let total = Object.keys(correct).length;

    for (const key in correct) {
      const selected = document.querySelector('input[name="' + key + '"]:checked');
      const feedback = document.getElementById(key + '-feedback');
      if (!selected) {
        feedback.textContent = 'Please choose an answer.';
        feedback.style.color = '#b36b00';
      } else if (selected.value === correct[key]) {
        feedback.textContent = 'Correct!';
        feedback.style.color = 'green';
        score++;
      } else {
        feedback.textContent = 'Not quite. Check the article again.';
        feedback.style.color = 'red';
      }
    }
    document.getElementById('mc-score').textContent = 'Score: ' + score + ' / ' + total;
  }

  // --- Vocab MC ---
  function checkVocabMC() {
    const correct = {
      vm1: 'B',
      vm2: 'B',
      vm3: 'A',
      vm4: 'B',
      vm5: 'B'
    };
    let score = 0;
    let total = Object.keys(correct).length;

    for (const key in correct) {
      const selected = document.querySelector('input[name="' + key + '"]:checked');
      const feedback = document.getElementById(key + '-feedback');
      if (!selected) {
        feedback.textContent = 'Please choose an answer.';
        feedback.style.color = '#b36b00';
      } else if (selected.value === correct[key]) {
        feedback.textContent = 'Correct!';
        feedback.style.color = 'green';
        score++;
      } else {
        feedback.textContent = 'Not quite. Think about how the word is used in the article.';
        feedback.style.color = 'red';
      }
    }
    document.getElementById('vm-score').textContent = 'Score: ' + score + ' / ' + total;
  }

  // --- Cloze ---
  function checkCloze() {
    const inputs = document.querySelectorAll('.cloze-input');
    let score = 0;
    inputs.forEach(function(input) {
      const correct = input.dataset.answer.trim().toLowerCase();
      const value = input.value.trim().toLowerCase();
      if (!value) {
        input.style.backgroundColor = '#fff3cd'; // yellow (empty)
      } else if (value === correct) {
        input.style.backgroundColor = '#d4edda'; // green
        score++;
      } else {
        input.style.backgroundColor = '#f8d7da'; // red
      }
    });
    document.getElementById('cloze-score').textContent = 'Score: ' + score + ' / ' + inputs.length;
  }

  // --- Matching ---
  function checkMatching() {
    const selects = document.querySelectorAll('.match-select');
    let score = 0;
    selects.forEach(function(sel) {
      const correct = sel.dataset.answer;
      if (!sel.value) {
        sel.style.backgroundColor = '#fff3cd'; // empty
      } else if (sel.value === correct) {
        sel.style.backgroundColor = '#d4edda';
        score++;
      } else {
        sel.style.backgroundColor = '#f8d7da';
      }
    });
    document.getElementById('match-score').textContent = 'Score: ' + score + ' / ' + selects.length;
  }

  // --- Reorder ---
  function checkReorder() {
    const items = document.querySelectorAll('.reorder-input');
    let score = 0;
    let total = items.length;
    items.forEach(function(input) {
      const correctOrder = parseInt(input.dataset.order, 10);
      const value = parseInt(input.value, 10);
      if (!value) {
        input.style.backgroundColor = '#fff3cd';
      } else if (value === correctOrder) {
        input.style.backgroundColor = '#d4edda';
        score++;
      } else {
        input.style.backgroundColor = '#f8d7da';
      }
    });
    document.getElementById('reorder-score').textContent = 'Score: ' + score + ' / ' + total;
  }

  // --- PRONUNCIATION PRACTICE ---

  // Text-to-speech
  function speakWord(word) {
    const utter = new SpeechSynthesisUtterance(word);
    utter.lang = "en-US";
    speechSynthesis.speak(utter);
  }

  let recognition = null;
  let activePronTarget = null;
  let activePronFeedback = null;

  function initRecognition() {
    if (!('webkitSpeechRecognition' in window)) {
      return null;
    }
    if (!recognition) {
      recognition = new webkitSpeechRecognition();
      recognition.lang = "en-US";
      recognition.interimResults = false;
      recognition.maxAlternatives = 1;

      recognition.onresult = function(event) {
        if (!activePronFeedback || !activePronTarget) return;
        const result = event.results[0][0].transcript.toLowerCase().trim();
        const correct = activePronTarget.toLowerCase().trim();

        if (result === correct) {
          activePronFeedback.textContent = "✔ Good! You said: \"" + result + "\" (correct)";
          activePronFeedback.style.color = "green";
        } else {
          activePronFeedback.textContent = "✘ You said: \"" + result +
            "\". Target word: \"" + activePronTarget + "\". Try again!";
          activePronFeedback.style.color = "red";
        }

        activePronTarget = null;
        activePronFeedback = null;
      };

      recognition.onerror = function() {
        if (activePronFeedback) {
          activePronFeedback.textContent = "Error capturing audio. Please try again.";
          activePronFeedback.style.color = "red";
        }
        activePronTarget = null;
        activePronFeedback = null;
      };
    }
    return recognition;
  }

  function startRecognitionPron(button, targetWord) {
    const rec = initRecognition();
    const feedback = button.parentElement.querySelector('.feedback');

    if (!rec) {
      feedback.textContent = "Speech recognition is not supported in this browser. Please use Chrome or Edge.";
      feedback.style.color = "red";
      return;
    }

    activePronTarget = targetWord;
    activePronFeedback = feedback;
    feedback.textContent = "Listening...";
    feedback.style.color = "#555";

    rec.start();
  }
